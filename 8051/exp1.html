W=(Y+3Z-6X)/6D

Where D=03H, X=02H, Y=25H and Z=12H
MOV A,#25H ; Move Y into accumulator
ADD A,#30H ; Add 3Z to accumulator
MOV R2,A ; Move result to R2
MOV A,#02H ; Move X into accumulator
SUBB A,R2 ; Subtract 6X from accumulator
MOV R3,A ; Move result to R3
MOV A,#03H ; Move D into accumulator
DIV AB ; Divide accumulator by B (D)
MOV R4,A ; Move quotient to R4
